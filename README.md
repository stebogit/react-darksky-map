### Installation

    $ npm install react-darksky-map --save
  

### Usage

```jsx harmony

import DarkskyMap from 'react-darksky-map';

<DarkskyMap lat={47.591} lng={-122.380} zoom={4} mapField="precip" url="http://example.com" apiKey="your-darksky-api-key" />

```
